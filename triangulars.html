  2. (2,5 punts) Ha d’utilitzar una funció de creació pròpia per comprovar si el nombre és triangular o no. Ha d’indicar a la consola el resultat d’aquesta funció.

    Un nombre triangular és el resultat de sumar els n primers nombres naturals. S'anomenen d'aquesta manera perquè són el nombre d'elements necessaris per crear un triangle equilàter.

    Per comprovar si un nombre x és triangular hem de trobar un valor n tal que

    1, 3, 6, 10, 15, 21, ... són nombres triangulars.

                    
function esTriangular(sencer){
    //Passam el nombre rebut com a String a int
    let i = parseInt(sencer);
    //Cream una variable boleana per definir si es triangular
    let bolea = false;
    //Bucle per cercar si correspon el nombre donat amb els nombres triangulars
    for(let b = 1; b < i; b++){
        //Algorisme per comprovar triangular i mostram el resultat
        let resultat = b*(b+1)/2;
        //Mostram cada un dels resultats del bucle
        console.log("Resultat: "+ resultat);
        //Si hi ha coincidencia aturam el bucle i canviam el bolea a true
        if(i === resultat){
            bolea = true;
            break;
        //Si el resultat del bucle es major al nombre facilitat aturam el bucle
        }else if(resultat > i){
        break;
        }
    }
    //Si el bolea es true
    if(bolea){
        console.log("El nombre " + i + " SI es triangular");
    //si el bolea es false
    }else{
    console.log("El nombre " + i + " No es triangular")
    }
}
